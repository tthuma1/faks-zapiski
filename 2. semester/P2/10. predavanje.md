import java.util.HashSet;  
  
public class Dnevi {  
    public static void main(String[] args) {  
        HashSet<String> dnevi = new HashSet<>();  
  
        dnevi.add("PON");  
        dnevi.add("TOR");  
        dnevi.add("SRE");  
  
        System.out.println(dnevi.size());  
        System.out.println(dnevi.contains("PON"));  
  
        dnevi.remove("PON");  
        System.out.println(dnevi.size());  
    }  
}
